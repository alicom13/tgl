Contoh Kode Penggunaan:


// 1. Inisialisasi dasar
$('#tanggal1').tgl();

// 2. Dengan callback events
$('#tanggal2').tgl({
    onValid: function(value) {
        console.log('Tanggal valid:', value);
    },
    onInvalid: function(value) {
        console.log('Tanggal tidak valid:', value);
    },
    onChange: function(value, isValid) {
        console.log('Perubahan:', value, isValid);
    }
});

// 3. Dengan validasi kustom
$('#tanggal3').tgl({
    minYear: 2020,
    maxYear: 2030
});

// 4. Multiple elements
$('.tanggal-range').tgl();

// Memanggil method
var dbFormat = $('#tanggal1').tgl('toDatabaseFormat');
var isValid = $('#tanggal1').tgl('validate');
$('#tanggal1').tgl('setValue', '31/12/2023');
$('#tanggal1').tgl('clear');

// Method statis
var isValid = $.tgl.isValid('31/12/2023');
var dbFormat = $.tgl.toDatabaseFormat('31/12/2023');
var today = $.tgl.today();

// Destroy plugin
$('#tanggal1').tgl('destroy');
