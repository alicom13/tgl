Contoh Kode Penggunaan:


// 1. Penggunaan dasar
const tanggal1 = new DateInput(document.getElementById('tanggal1'));

// 2. Dengan callback events
const tanggal2 = new DateInput(document.getElementById('tanggal2'), {
    onValid: (value) => {
        console.log('Tanggal valid:', value);
    },
    onInvalid: (value) => {
        console.log('Tanggal tidak valid:', value);
    },
    onChange: (value, isValid) => {
        console.log('Perubahan:', value, isValid);
    }
});

// 3. Dengan validasi kustom
const tanggal3 = new DateInput(document.getElementById('tanggal3'), {
    minYear: 2020,
    maxYear: 2030
});

// Method yang tersedia:
// - tanggal1.toDatabaseFormat() → "yyyy-mm-dd"
// - tanggal1.toDateObject()     → Date object
// - tanggal1.getValue()         → "dd/mm/yyyy"
// - tanggal1.setValue("31/12/2023")
// - tanggal1.validate()         → boolean
// - tanggal1.clear()

// Method statis:
// - DateInput.isValid("31/12/2023") → boolean
// - DateInput.toDatabaseFormat("31/12/2023") → "2023-12-31"
// - DateInput.today() → tanggal hari ini "dd/mm/yyyy"
